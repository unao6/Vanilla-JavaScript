        비동기실행
        console.log('시작');
        setTimeout( ()=>{
            console.log('중간실행');
        },3000)

        console.log('마지막실행');
        //////// 시작>마지막실행>중간실행 나타남.

        그럼 해결하려면 어떻게 할까?

        1. 옛날에 억지로 하던 버전
        console.log('시작');
        function make( callback) {
            setTimeout( ()=>{
            console.log('중간실행');
            callback();
            },3000)
        }        
        make( ()=> console.log('마지막실행') );

        2. callback hell. 이것또한 예전작업
        console.log('시작');
        setTimeout( () => {
            console.log('1번 실행');
            setTimeout( () => {
                console.log('2번 실행');
                setTimeout( () => {
                    console.log('3번 실행');
                        setTimeout( () => {
                        console.log('4번 실행');
                    },2000);
                },2000);
            },2000);
        },2000);
